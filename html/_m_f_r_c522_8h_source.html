58 enum PCD_Register {
59 // Page 0: Command and status
60 // 0x00 // reserved for future use
61 CommandReg = 0x01 << 1, // starts and stops command execution
62 ComIEnReg = 0x02 << 1, // enable and disable interrupt request control bits
63 DivIEnReg = 0x03 << 1, // enable and disable interrupt request control bits
64 ComIrqReg = 0x04 << 1, // interrupt request bits
65 DivIrqReg = 0x05 << 1, // interrupt request bits
66 ErrorReg = 0x06 << 1, // error bits showing the error status of the last command executed
67 Status1Reg = 0x07 << 1, // communication status bits
68 Status2Reg = 0x08 << 1, // receiver and transmitter status bits
69 FIFODataReg = 0x09 << 1, // input and output of 64 byte FIFO buffer
70 FIFOLevelReg = 0x0A << 1, // number of bytes stored in the FIFO buffer
71 WaterLevelReg = 0x0B << 1, // level for FIFO underflow and overflow warning
72 ControlReg = 0x0C << 1, // miscellaneous control registers
73 BitFramingReg = 0x0D << 1, // adjustments for bit-oriented frames
74 CollReg = 0x0E << 1, // bit position of the first bit-collision detected on the RF interface
75 // 0x0F // reserved for future use
76
77 // Page 1: Command
78 // 0x10 // reserved for future use
79 ModeReg = 0x11 << 1, // defines general modes for transmitting and receiving
80 TxModeReg = 0x12 << 1, // defines transmission data rate and framing
81 RxModeReg = 0x13 << 1, // defines reception data rate and framing
82 TxControlReg = 0x14 << 1, // controls the logical behavior of the antenna driver pins TX1 and TX2
83 TxASKReg = 0x15 << 1, // controls the setting of the transmission modulation
84 TxSelReg = 0x16 << 1, // selects the internal sources for the antenna driver
85 RxSelReg = 0x17 << 1, // selects internal receiver settings
86 RxThresholdReg = 0x18 << 1, // selects thresholds for the bit decoder
87 DemodReg = 0x19 << 1, // defines demodulator settings
88 // 0x1A // reserved for future use
89 // 0x1B // reserved for future use
90 MfTxReg = 0x1C << 1, // controls some MIFARE communication transmit parameters
91 MfRxReg = 0x1D << 1, // controls some MIFARE communication receive parameters
92 // 0x1E // reserved for future use
93 SerialSpeedReg = 0x1F << 1, // selects the speed of the serial UART interface
94
95 // Page 2: Configuration
96 // 0x20 // reserved for future use
97 CRCResultRegH = 0x21 << 1, // shows the MSB and LSB values of the CRC calculation
98 CRCResultRegL = 0x22 << 1,
99 // 0x23 // reserved for future use
100 ModWidthReg = 0x24 << 1, // controls the ModWidth setting?
101 // 0x25 // reserved for future use
102 RFCfgReg = 0x26 << 1, // configures the receiver gain
103 GsNReg = 0x27 << 1, // selects the conductance of the antenna driver pins TX1 and TX2 for modulation
104 CWGsPReg = 0x28 << 1, // defines the conductance of the p-driver output during periods of no modulation
105 ModGsPReg = 0x29 << 1, // defines the conductance of the p-driver output during periods of modulation
106 TModeReg = 0x2A << 1, // defines settings for the internal timer
107 TPrescalerReg = 0x2B << 1, // the lower 8 bits of the TPrescaler value. The 4 high bits are in TModeReg.

132 // MFRC522 commands. Described in chapter 10 of the datasheet.
133 enum PCD_Command {
134 PCD_Idle = 0x00, // no action, cancels current command execution
135 PCD_Mem = 0x01, // stores 25 bytes into the internal buffer
136 PCD_GenerateRandomID = 0x02, // generates a 10-byte random ID number
137 PCD_CalcCRC = 0x03, // activates the CRC coprocessor or performs a self test
138 PCD_Transmit = 0x04, // transmits data from the FIFO buffer
139 PCD_NoCmdChange = 0x07, // no command change, can be used to modify the CommandReg register bits without affecting the command, for example, the PowerDown bit
140 PCD_Receive = 0x08, // activates the receiver circuits
141 PCD_Transceive = 0x0C, // transmits data from FIFO buffer to antenna and automatically activates the receiver after transmission
142 PCD_MFAuthent = 0x0E, // performs the MIFARE standard authentication as a reader
143 PCD_SoftReset = 0x0F // resets the MFRC522
144 };

162 // Commands sent to the PICC.
163 enum PICC_Command {
164 // The commands used by the PCD to manage communication with several PICCs (ISO 14443-3, Type A, section 6.4)
165 PICC_CMD_REQA = 0x26, // REQuest command, Type A. Invites PICCs in state IDLE to go to READY and prepare for anticollision or selection. 7 bit frame.
166 PICC_CMD_WUPA = 0x52, // Wake-UP command, Type A. Invites PICCs in state IDLE and HALT to go to READY(*) and prepare for anticollision or selection. 7 bit frame.
167 PICC_CMD_CT = 0x88, // Cascade Tag. Not really a command, but used during anti collision.
168 PICC_CMD_SEL_CL1 = 0x93, // Anti collision/Select, Cascade Level 1
169 PICC_CMD_SEL_CL2 = 0x95, // Anti collision/Select, Cascade Level 2
170 PICC_CMD_SEL_CL3 = 0x97, // Anti collision/Select, Cascade Level 3
171 PICC_CMD_HLTA = 0x50, // HaLT command, Type A. Instructs an ACTIVE PICC to go to state HALT.
172 // The commands used for MIFARE Classic (from http://www.nxp.com/documents/data_sheet/MF1S503x.pdf, Section 9)
173 // Use PCD_MFAuthent to authenticate access to a sector, then use these commands to read/write/modify the blocks on the sector.
174 // The read/write commands can also be used for MIFARE Ultralight.
175 PICC_CMD_MF_AUTH_KEY_A = 0x60, // Perform authentication with Key A
176 PICC_CMD_MF_AUTH_KEY_B = 0x61, // Perform authentication with Key B
177 PICC_CMD_MF_READ = 0x30, // Reads one 16 byte block from the authenticated sector of the PICC. Also used for MIFARE Ultralight.
178 PICC_CMD_MF_WRITE = 0xA0, // Writes one 16 byte block to the authenticated sector of the PICC. Called "COMPATIBILITY WRITE" for MIFARE Ultralight.
179 PICC_CMD_MF_DECREMENT = 0xC0, // Decrements the contents of a block and stores the result in the internal data register.
180 PICC_CMD_MF_INCREMENT = 0xC1, // Increments the contents of a block and stores the result in the internal data register.
181 PICC_CMD_MF_RESTORE = 0xC2, // Reads the contents of a block into the internal data register.
182 PICC_CMD_MF_TRANSFER = 0xB0, // Writes the contents of the internal data register to a block.
183 // The commands used for MIFARE Ultralight (from http://www.nxp.com/documents/data_sheet/MF0ICU1.pdf, Section 8.6)
184 // The PICC_CMD_MF_READ and PICC_CMD_MF_WRITE can also be used for MIFARE Ultralight.
185 PICC_CMD_UL_WRITE = 0xA2 // Writes one 4 byte page to the PICC.
186 };
