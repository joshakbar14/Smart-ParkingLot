606// Defines for ST
607// GPIO register offsets from BCM2835_ST_BASE.
608// Offsets into the ST Peripheral block in bytes per 12.1 System Timer Registers
609// The System Timer peripheral provides four 32-bit timer channels and a single 64-bit free running counter.
610// BCM2835_ST_CLO is the System Timer Counter Lower bits register.
611// The system timer free-running counter lower register is a read-only register that returns the current value
612// of the lower 32-bits of the free running counter.
613// BCM2835_ST_CHI is the System Timer Counter Upper bits register.
614// The system timer free-running counter upper register is a read-only register that returns the current value
615// of the upper 32-bits of the free running counter.
616#define BCM2835_ST_CS 0x0000
617#define BCM2835_ST_CLO 0x0004
618#define BCM2835_ST_CHI 0x0008
